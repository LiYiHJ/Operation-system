    """🆕 解析文件（返回映射结果）"""

        filepath = save_file(file)

            'totalRows': result.total_rows,
            'mapping': result.data[:10],  # 前10行预览
            'errors': result.errors,
            'warnings': result.warnings
        })
    except Exception as e:
        return jsonify({'error': str(e)}), 500
